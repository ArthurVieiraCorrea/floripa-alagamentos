```js
import { defineConfig } from 'vite';
import { resolve } from 'path';

export default defineConfig({
  server: {
    port: 5173,
    proxy: {
      '/api':  'http://localhost:3001',
      '/auth': 'http://localhost:3001'
    }
  },
  build: {
    rollupOptions: {
      input: {
        index: resolve(__dirname, 'src/pages/landing.html'),  // serve '/' — landing page
        app:   resolve(__dirname, 'index.html')               // serve '/app' — app existente
      }
    }
  }
});
```

Explicação: A chave `index` gera `dist/index.html` (landing). A chave `app` gera `dist/app.html` (app existente, entry point inalterado em frontend/index.html).

**2. Modificar backend/src/app.js** — substituir o bloco de linhas 86-97 (express.static + catch-all):

Localizar no arquivo:
```js
// Serve o frontend em produção
const frontendDist = path.join(__dirname, '../../frontend/dist');
const fs = require('fs');
app.use(express.static(frontendDist));
app.get('*', (req, res) => {
  const indexPath = path.join(frontendDist, 'index.html');
  if (fs.existsSync(indexPath)) {
    res.sendFile(indexPath);
  } else {
    res.status(404).json({ erro: 'Frontend não compilado. Execute: cd frontend && npm run build' });
  }
});
```

Substituir por:
```js
// Serve o frontend em produção
const frontendDist = path.join(__dirname, '../../frontend/dist');
const fs = require('fs');
app.use(express.static(frontendDist));

// Rota explícita para o app — DEVE vir antes do catch-all GET *
// Após build multi-page, o app está em dist/app.html (não dist/index.html)
app.get('/app', (req, res) => {
  const appPath = path.join(frontendDist, 'app.html');
  if (fs.existsSync(appPath)) {
    res.sendFile(appPath);
  } else {
    res.status(404).json({ erro: 'App não compilado. Execute: cd frontend && npm run build' });
  }
});

// Catch-all serve a landing page (dist/index.html após build multi-page)
app.get('*', (req, res) => {
  const indexPath = path.join(frontendDist, 'index.html');
  if (fs.existsSync(indexPath)) {
    res.sendFile(indexPath);
  } else {
    res.status(404).json({ erro: 'Frontend não compilado. Execute: cd frontend && npm run build' });
  }
});
```

IMPORTANTE: `app.get('/app', ...)` deve ficar IMEDIATAMENTE após `app.use(express.static(...))` e ANTES do `app.get('*', ...)`. A ordem é crítica — Express processa rotas sequencialmente.
