
**buildUrl() — janela de 24h em UTC-3:**
```javascript
function buildUrl() {
  const KEY = process.env.VISUAL_CROSSING_API_KEY;
  const now = new Date();
  const yesterday = new Date(now.getTime() - 24 * 60 * 60 * 1000);
  const toLocalDate = d => {
    const local = new Date(d.getTime() - 3 * 60 * 60 * 1000); // UTC-3 fixo
    return local.toISOString().substring(0, 10); // YYYY-MM-DD
  };
  const date1 = toLocalDate(yesterday);
  const date2 = toLocalDate(now);
  return [
    'https://weather.visualcrossing.com/VisualCrossingWebServices/rest/services/timeline',
    `/${LAT},${LON}/${date1}/${date2}`,
    '?unitGroup=metric',
    '&include=hours',
    '&elements=datetime,precip,preciptype',
    `&key=${KEY}`
  ].join('');
}
```

```javascript
const db = getDb();
let totalMm = 0;
db.run('BEGIN');
try {
  for (const day of data.days ?? []) {
    for (const hour of day.hours ?? []) {
      const mm = hour.precip ?? 0;
      // Combinar "2026-04-22" + "T" + "01:00:00".substring(0,5) = "2026-04-22T01:00"
      const observed_time = `${day.datetime}T${hour.datetime.substring(0, 5)}`;
      totalMm += mm;
      db.run(
        `INSERT OR IGNORE INTO weather_observations
           (bairro, observed_time, precipitacao_mm, fonte, fetched_at)
         VALUES ('florianopolis', ?, ?, 'visual-crossing', datetime('now'))`,
        [observed_time, mm]
      );
    }
  }
  db.run('COMMIT');
} catch (dbErr) {
  try { db.run('ROLLBACK'); } catch (_) {}
  console.error('[visual-crossing] Erro ao persistir observacoes:', dbErr.message);
  return;
}
```

| Boundary | Description |
|----------|-------------|
| Visual Crossing API → Node.js fetch | Resposta JSON de API externa parseada e persistida no SQLite |
| process.env → buildUrl() | API key lida de variavel de ambiente e inserida na URL — nunca logada |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-09.1-02-01 | Information Disclosure | VISUAL_CROSSING_API_KEY na URL do fetch | mitigate | Chave aparece apenas na URL construida internamente em buildUrl(); nunca logada via console.*; URL nao e armazenada no banco ou exposta em responses HTTP |
| T-09.1-02-02 | Tampering | Resposta JSON maliciosa do Visual Crossing | mitigate | Todos os campos numericos usam `?? 0` para default; campos de string (observed_time) sao construidos a partir de day.datetime e hour.datetime via concatenacao de template literal — sem eval(); INSERT usa parametros preparados db.run(sql, [params]) sem concatenacao SQL |
